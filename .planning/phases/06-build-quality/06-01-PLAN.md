  <action>
Create `eslint.config.mjs` using ESLint 9 flat config format with Next.js configuration.

Use the pattern from research (Next.js official docs):
```javascript
import { dirname } from 'path'
import { fileURLToPath } from 'url'
import { FlatCompat } from '@eslint/eslintrc'

const __filename = fileURLToPath(import.meta.url)
const __dirname = dirname(__filename)

const compat = new FlatCompat({
  baseDirectory: __dirname,
})

const eslintConfig = [
  ...compat.extends('next/core-web-vitals', 'next/typescript'),
  {
    rules: {
      // Migrated from .eslintrc.json
      '@typescript-eslint/no-unused-vars': ['error', { argsIgnorePattern: '^_' }],
      '@typescript-eslint/explicit-function-return-type': 'off',
      '@typescript-eslint/no-explicit-any': 'warn',
      'no-console': ['warn', { allow: ['warn', 'error', 'info'] }],
      'no-eval': 'error',
      'no-implied-eval': 'error',
      '@next/next/no-img-element': 'error',
      'react/no-unknown-property': 'off',
      'react/prop-types': 'off',
    },
  },
  {
    ignores: [
      '.next/**',
      'out/**',
      'build/**',
      'node_modules/**',
      'next-env.d.ts',
    ],
  },
]

export default eslintConfig
```

Note: Use FlatCompat from @eslint/eslintrc to bridge legacy config patterns. Install if needed: `pnpm add -D @eslint/eslintrc`
