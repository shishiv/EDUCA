```typescript
const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault()

  if (!formData.descricao.trim()) {
    toast.error('Preencha a descricao da vivencia')
    return
  }

  if (formData.campos_experiencia.length === 0) {
    toast.error('Selecione pelo menos um campo de experiencia')
    return
  }

  setLoading(true)

  try {
    const response = await fetch('/api/vivencias', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        aluno_id: alunoId,
        turma_id: formData.turma_id, // Get from student's active enrollment or form
        data_vivencia: formData.data_vivencia,
        campos_experiencia: formData.campos_experiencia,
        descricao: formData.descricao,
        observacoes: formData.observacoes || null,
        escopo: formData.escopo || 'individual'
      })
    })

    if (!response.ok) {
      const errorData = await response.json()
      throw new Error(errorData.error || 'Erro ao salvar vivencia')
    }

    toast.success('Vivencia registrada com sucesso!')
    router.push(`/dashboard/alunos/${alunoId}/diario`)
  } catch (err) {
    console.error('Error saving vivencia:', err)
    toast.error(err instanceof Error ? err.message : 'Erro ao salvar vivencia')
  } finally {
    setLoading(false)
  }
}
```

```typescript
  async function loadData() {

        .from('alunos')
        .select('id, nome_completo, data_nascimento')
        .eq('id', alunoId)
        .single()

      if (studentError) throw studentError
      setStudent(studentData)

      // Load vivencias from API
      const response = await fetch(`/api/vivencias?aluno_id=${alunoId}`)

      if (!response.ok) {
        const errorData = await response.json()
        throw new Error(errorData.error || 'Erro ao carregar vivencias')
      }

      const { data: vivenciasData } = await response.json()
      setVivencias(vivenciasData || [])

    } catch (err) {
      console.error('Error loading data:', err)
      setError(err instanceof Error ? err.message : 'Erro ao carregar dados')
    } finally {
      setLoading(false)
    }
  }

  loadData()
}, [alunoId])
```
