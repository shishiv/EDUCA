```typescript
async function getOrCreateSession(turmaId: string, date: Date) {
  const dateStr = format(date, 'yyyy-MM-dd')

  // Check for existing session
  const { data: existing } = await supabase
    .from('sessoes_aula')
    .select('id, status')
    .eq('turma_id', turmaId)
    .eq('data_aula', dateStr)
    .single()

  if (existing) return existing

  // Create new session if none exists
  const { data: newSession, error } = await supabase
    .from('sessoes_aula')
    .insert({
      turma_id: turmaId,
      data_aula: dateStr,
      status: 'ABERTA'
    })
    .select()
    .single()

  if (error) throw error
  return newSession
}

import { AttendanceImmutabilityService } from '@/lib/services/attendance-immutability'

const immutabilityService = new AttendanceImmutabilityService()

async function handleSave() {
  setIsSaving(true)
  try {
    // Check lock status
    const lockInfo = await immutabilityService.getSessionLockInfo(sessionId, currentDate)
    if (lockInfo.isLocked) {
      toast.error(`Chamada travada: ${lockInfo.lockReason}`)
      setIsLocked(true)
      return
    }

    // Proceed with batch save
    const records = Array.from(attendance.entries()).map(([studentId, record]) => ({
      matricula_id: studentId, // May need to map student to matricula
      status_presenca: record.status,
      justificativa: record.justificativa || null,
      presente: record.status !== 'F' // P and J count as present
    }))

    // Use existing attendance API or direct supabase
    const { error } = await supabase
      .from('frequencia')
      .upsert(records.map(r => ({
        sessao_id: sessionId,
        ...r,
        data_aula: format(currentDate, 'yyyy-MM-dd'),
        marcado_em: new Date().toISOString()
      })))

    if (error) throw error

    toast.success('Chamada salva com sucesso!')
    setOriginalAttendance(new Map(attendance))
  } catch (error: any) {
    toast.error(`Erro ao salvar: ${error.message}`)
  } finally {
    setIsSaving(false)
  }
}
```

```typescript
async function loadAttendanceForDate(date: Date) {
  const dateStr = format(date, 'yyyy-MM-dd')

  const { data: session } = await supabase
    .from('sessoes_aula')
    .select('id')
    .eq('turma_id', turmaId)
    .eq('data_aula', dateStr)
    .single()

  if (!session) {
    // No session for this date - initialize all as Present if today
    if (isSameDay(date, new Date())) {
      initializeAllPresent()
    } else {
      setAttendance(new Map())
    }
    return
  }

  const { data: frequencias } = await supabase
    .from('frequencia')
    .select('matricula_id, status_presenca, justificativa')
    .eq('sessao_id', session.id)

  const attendanceMap = new Map()
  frequencias?.forEach(f => {
    attendanceMap.set(f.matricula_id, {
      status: f.status_presenca,
      justificativa: f.justificativa
    })
  })

  setAttendance(attendanceMap)
  setOriginalAttendance(new Map(attendanceMap))
}
```
