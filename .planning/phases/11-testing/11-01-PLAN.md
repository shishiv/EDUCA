<task type="auto">

  <files>gestao_fronteira/vitest.config.mts</files>
  <action>
Create vitest.config.mts in gestao_fronteira/ root:

```typescript
import { defineConfig } from 'vitest/config'
import react from '@vitejs/plugin-react'
import tsconfigPaths from 'vite-tsconfig-paths'

export default defineConfig({
  plugins: [tsconfigPaths(), react()],
  test: {
    environment: 'jsdom',
    globals: true,
    setupFiles: ['./tests/vitest.setup.ts'],
    include: ['tests/unit/**/*.test.ts', 'tests/unit/**/*.test.tsx'],
    exclude: ['node_modules', 'tests/e2e'],
    coverage: {
      provider: 'v8',
      reporter: ['text', 'json', 'html'],
      include: ['lib/services/**', 'lib/validation/**'],
      exclude: ['**/*.d.ts', '**/*.test.ts'],
    },
  },
})
```

Key configuration:
- `tsconfigPaths()` enables @/* path aliases
- `environment: 'jsdom'` for DOM testing
- `globals: true` enables describe/it/expect without imports
- Coverage focused on lib/services and lib/validation (critical code)

```typescript
import { vi } from 'vitest'
import '@testing-library/jest-dom/vitest'

// Mock next/headers - prevents "cookies() can only be called from Server Component" errors
vi.mock('next/headers', () => ({
  cookies: vi.fn(() => ({
    get: vi.fn(),
    set: vi.fn(),
    delete: vi.fn(),
    getAll: vi.fn(() => []),
  })),
  headers: vi.fn(() => new Map()),
}))

// Mock Supabase client - prevents database calls in unit tests
vi.mock('@/lib/supabase', () => ({
  supabase: {
    from: vi.fn(() => ({
      select: vi.fn().mockReturnThis(),
      insert: vi.fn().mockReturnThis(),
      update: vi.fn().mockReturnThis(),
      delete: vi.fn().mockReturnThis(),
      eq: vi.fn().mockReturnThis(),
      neq: vi.fn().mockReturnThis(),
      or: vi.fn().mockReturnThis(),
      and: vi.fn().mockReturnThis(),
      order: vi.fn().mockReturnThis(),
      limit: vi.fn().mockReturnThis(),
      single: vi.fn().mockResolvedValue({ data: null, error: null }),
      maybeSingle: vi.fn().mockResolvedValue({ data: null, error: null }),
    })),
    auth: {
      getUser: vi.fn().mockResolvedValue({ data: { user: null }, error: null }),
      getSession: vi.fn().mockResolvedValue({ data: { session: null }, error: null }),
    },
    rpc: vi.fn().mockResolvedValue({ data: null, error: null }),
  },
}))

// Suppress console output during tests (optional, uncomment if noisy)
// vi.spyOn(console, 'log').mockImplementation(() => {})
// vi.spyOn(console, 'warn').mockImplementation(() => {})
```

This setup:
1. Imports jest-dom matchers for Vitest
2. Mocks next/headers to prevent Server Component errors
3. Mocks Supabase client completely - all database calls are isolated
