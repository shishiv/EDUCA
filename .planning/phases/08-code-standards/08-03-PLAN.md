```typescript
async getFrequenciaReport(filters: FrequenciaFilters): Promise<FrequenciaReportData[]> {
  try {
    const { data, error } = await supabase
      .from('frequencia')
      .select(`
        *,
        aluno:alunos(nome, cpf),
        turma:turmas(nome)
      `)
      .gte('data', filters.startDate)
      .lte('data', filters.endDate)

    if (error) {
      logger.error('Error fetching frequencia report', error, {
        feature: 'reports',
        action: 'get_frequencia_report'
      })
      throw error
    }

    return (data || []) as FrequenciaReportData[]
  } catch (error) {
    logger.error('Error in getFrequenciaReport', error, { feature: 'reports' })
    throw error
  }
}
```
