**Step 2: For each diario page, replace inline queries**

Pattern to follow:

```typescript
// BEFORE
const { data, error } = await supabase
  .from('vivencias')
  .select('*')
  .eq('aluno_id', alunoId)

if (error) console.error('Error:', error)

// AFTER
import { vivenciasApi } from '@/lib/api/vivencias'
import { logger } from '@/lib/logger'

try {
  const vivencias = await vivenciasApi.getByAluno(alunoId)
  setVivencias(vivencias)
} catch (error) {
  logger.error('Error loading vivencias', error, {
    feature: 'diario',
    action: 'load_vivencias',
    metadata: { alunoId }
  })
  toast.error('Erro ao carregar vivencias')
}
```
