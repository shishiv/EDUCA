/**
 * Get help tips for a given pathname
 * Falls back to default tips if no specific tips exist
 */
export function getHelpTipsForRoute(pathname: string): HelpTip[] {
  // Check for exact match first
  if (helpContent[pathname]) {
    return helpContent[pathname]
  }

  // Check for partial match (e.g., /dashboard/turmas/123 matches /dashboard/turmas)
  for (const route of Object.keys(helpContent)) {
    if (pathname.startsWith(route) && route !== 'default') {
      return helpContent[route]
    }
  }

  return helpContent.default
}

**HelpButton.tsx** - Help icon for header per CONTEXT.md:
```tsx
'use client'

import * as React from 'react'
import { usePathname } from 'next/navigation'
import { HelpCircle } from 'lucide-react'
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from '@/components/ui/popover'
import { getHelpTipsForRoute } from './help-content'

export function HelpButton() {
  const pathname = usePathname()
  const [isOpen, setIsOpen] = React.useState(false)

  const tips = getHelpTipsForRoute(pathname)

  return (
    <Popover open={isOpen} onOpenChange={setIsOpen}>
      <PopoverTrigger asChild>
        <button
          type="button"
          className="h-10 w-10 rounded-[10px] bg-gray-50 hover:bg-gray-100 border border-gray-200 flex items-center justify-center text-gray-600 hover:text-gray-800 transition-all duration-200"
          aria-label="Ajuda"
          title="Ajuda"
        >
          <HelpCircle className="h-5 w-5" />
        </button>
      </PopoverTrigger>
      <PopoverContent className="w-80" align="end">
        <div className="space-y-4">
          <h3 className="font-semibold text-sm text-green-600">Dicas desta pagina</h3>
          {tips.map((tip, i) => (
            <div key={i} className="space-y-1">
              <h4 className="font-medium text-sm text-gray-900">{tip.title}</h4>
              <p className="text-sm text-muted-foreground">{tip.content}</p>
            </div>
          ))}
        </div>
      </PopoverContent>
    </Popover>
  )
}
```
