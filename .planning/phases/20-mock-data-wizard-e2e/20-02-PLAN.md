/**
 * First Access Context - Onboarding Wizard State
 *
 * Shows first-access wizard for users with primeiro_login=true.
 * Follows DemoModeContext pattern with sessionStorage hydration.
 *
 * Per CONTEXT.md:
 * - "Recomendado mas pulavel" - Skip button visible
 * - Wizard reaparece until completed (but can be dismissed per session)
 * - After completion, primeiro_login set to false in database
 */

'use client'

import * as React from 'react'
import { useAuth } from '@/hooks/use-auth'
import { supabase } from '@/lib/supabase'
import { logger } from '@/lib/logger'

const STORAGE_KEY = 'gestao-fronteira-wizard-dismissed'

type UserProfile = 'admin' | 'diretor' | 'secretario' | 'professor' | 'responsavel'

interface FirstAccessContextType {
  showWizard: boolean
  currentStep: number
  totalSteps: number
  userProfile: UserProfile | null
  nextStep: () => void
  prevStep: () => void
  skipWizard: () => void
  completeWizard: () => Promise<void>
}

const FirstAccessContext = React.createContext<FirstAccessContextType | null>(null)

interface FirstAccessProviderProps {
  children: React.ReactNode
}

export function FirstAccessProvider({ children }: FirstAccessProviderProps) {
  const { userProfile, user } = useAuth()

  const [showWizard, setShowWizard] = React.useState(false)
  const [currentStep, setCurrentStep] = React.useState(0)
  const [hydrated, setHydrated] = React.useState(false)

  // Get user type for profile-specific steps
  const profileType = (userProfile?.tipo_usuario as UserProfile) || null

  // Calculate total steps based on profile
  const totalSteps = React.useMemo(() => {
    // Welcome + Profile-specific + Complete = 3 steps for all profiles
    return 3
  }, [])

  // Hydrate from sessionStorage after mount
  React.useEffect(() => {
    if (typeof window !== 'undefined') {
      const dismissed = sessionStorage.getItem(STORAGE_KEY)
      const shouldShow = userProfile?.primeiro_login === true && !dismissed
      setShowWizard(shouldShow)
      setHydrated(true)
    }
  }, [userProfile?.primeiro_login])

  const nextStep = React.useCallback(() => {
    setCurrentStep(prev => Math.min(prev + 1, totalSteps - 1))
  }, [totalSteps])

  const prevStep = React.useCallback(() => {
    setCurrentStep(prev => Math.max(prev - 1, 0))
  }, [])

  const skipWizard = React.useCallback(() => {
    setShowWizard(false)
    if (typeof window !== 'undefined') {
      sessionStorage.setItem(STORAGE_KEY, 'true')
    }
  }, [])

  const completeWizard = React.useCallback(async () => {
    if (!user?.id) return

    try {
      // Update primeiro_login in database
      const { error } = await supabase
        .from('users')
        .update({ primeiro_login: false })
        .eq('id', user.id)

      if (error) {
        logger.error('first-access', 'complete_wizard', error)
      }

      setShowWizard(false)
      // Also mark as dismissed in session to prevent flash on next navigation
      if (typeof window !== 'undefined') {
        sessionStorage.setItem(STORAGE_KEY, 'true')
      }
    } catch (err) {
      logger.error('first-access', 'complete_wizard', err instanceof Error ? err : new Error(String(err)))
    }
  }, [user?.id])

  const contextValue: FirstAccessContextType = {
    showWizard: hydrated && showWizard,
    currentStep,
    totalSteps,
    userProfile: profileType,
    nextStep,
    prevStep,
    skipWizard,
    completeWizard,
  }

  return (
    <FirstAccessContext.Provider value={contextValue}>
      {children}
    </FirstAccessContext.Provider>
  )
}

export function useFirstAccess() {
  const context = React.useContext(FirstAccessContext)
  if (!context) {
    throw new Error('useFirstAccess must be used within a FirstAccessProvider')
  }
  return context
}

**FirstAccessWizard.tsx** - Main wizard dialog component:
```tsx
'use client'

import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
  DialogFooter,
} from '@/components/ui/dialog'
import { Button } from '@/components/ui/button'
import { Progress } from '@/components/ui/progress'
import { useFirstAccess } from '@/contexts/first-access-context'
import { WelcomeStep } from './steps/WelcomeStep'
import { AdminStep } from './steps/AdminStep'
import { DiretorStep } from './steps/DiretorStep'
import { SecretarioStep } from './steps/SecretarioStep'
import { ProfessorStep } from './steps/ProfessorStep'
import { ChevronLeft, ChevronRight, X } from 'lucide-react'

export function FirstAccessWizard() {
  const {
    showWizard,
    currentStep,
    totalSteps,
    userProfile,
    nextStep,
    prevStep,
    skipWizard,
    completeWizard
  } = useFirstAccess()

  if (!showWizard) return null

  const progress = ((currentStep + 1) / totalSteps) * 100
  const isFirstStep = currentStep === 0
  const isLastStep = currentStep === totalSteps - 1

  // Get profile-specific step component
  const getProfileStep = () => {
    switch (userProfile) {
      case 'admin':
        return <AdminStep />
      case 'diretor':
        return <DiretorStep />
      case 'secretario':
        return <SecretarioStep />
      case 'professor':
        return <ProfessorStep />
      default:
        return <AdminStep /> // fallback
    }
  }

  // Render current step content
  const renderStep = () => {
    switch (currentStep) {
      case 0:
        return <WelcomeStep />
      case 1:
        return getProfileStep()
      case 2:
        return (
          <div className="text-center py-6">
            <div className="h-16 w-16 rounded-full bg-green-100 flex items-center justify-center mx-auto mb-4">
              <svg className="h-8 w-8 text-green-600" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M5 13l4 4L19 7" />
              </svg>
            </div>
            <h2 className="text-xl font-semibold text-gray-900 mb-2">Tudo pronto!</h2>
            <p className="text-gray-600">
              Voce pode acessar a ajuda a qualquer momento pelo icone de ajuda no menu superior.
            </p>
          </div>
        )
      default:
        return null
    }
  }

  return (
    <Dialog open={showWizard} onOpenChange={() => {}}>
      <DialogContent className="sm:max-w-lg" hideCloseButton>
        <DialogHeader>
          <div className="flex items-center justify-between">
            <DialogTitle className="text-lg">
              Primeiros Passos
            </DialogTitle>
            <Button
              variant="ghost"
              size="icon"
              onClick={skipWizard}
              className="h-8 w-8"
              title="Pular tutorial"
            >
              <X className="h-4 w-4" />
            </Button>
          </div>
          <DialogDescription className="sr-only">
            Tutorial de primeiro acesso ao sistema EDUCA
          </DialogDescription>
        </DialogHeader>

        {/* Progress indicator */}
        <div className="space-y-2">
          <Progress value={progress} className="h-1" />
          <p className="text-xs text-muted-foreground text-center">
            Passo {currentStep + 1} de {totalSteps}
          </p>
        </div>

        {/* Step content */}
        <div className="py-4">
          {renderStep()}
        </div>

        <DialogFooter className="flex-row justify-between sm:justify-between">
          <Button
            variant="outline"
            onClick={prevStep}
            disabled={isFirstStep}
            className="gap-1"
          >
            <ChevronLeft className="h-4 w-4" />
            Anterior
          </Button>

          {isLastStep ? (
            <Button onClick={completeWizard} className="gap-1 bg-green-600 hover:bg-green-700">
              Concluir
            </Button>
          ) : (
            <Button onClick={nextStep} className="gap-1">
              Proximo
              <ChevronRight className="h-4 w-4" />
            </Button>
          )}
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

import { QueryClient, QueryClientProvider } from '@tanstack/react-query'

    <QueryClientProvider client={queryClient}>
      <ServiceWorkerProvider>
        <FirstAccessProvider>
          {children}
        </FirstAccessProvider>
      </ServiceWorkerProvider>
    </QueryClientProvider>
  )
}
