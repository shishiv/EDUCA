/**
 * E2E Test Data Seeding Utilities
 *
 * Provides functions to seed and cleanup test data for E2E tests.
 * Uses Supabase service role key for direct database access.
 *
 * NOTE: These credentials should only be used in test environments.
 */

import { createClient } from '@supabase/supabase-js'

// Get Supabase credentials from environment
// E2E tests should have SUPABASE_SERVICE_ROLE_KEY available
const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL || ''
const supabaseServiceKey = process.env.SUPABASE_SERVICE_ROLE_KEY || process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY || ''

export const supabaseAdmin = createClient(supabaseUrl, supabaseServiceKey)

// Test data IDs - use valid UUID format for Supabase compatibility
export const TEST_IDS = {
  ESCOLA: 'e2e00000-0000-0000-0000-000000000001',
  TURMA: 'e2e00000-0000-0000-0000-000000000002',
  PROFESSOR_USER: 'e2e00000-0000-0000-0000-000000000003',
  ADMIN_USER: 'e2e00000-0000-0000-0000-000000000004',
} as const

/**
 * Seed a test escola
 */
export async function seedTestEscola() {
  const { data, error } = await supabaseAdmin
    .from('escolas')
    .upsert({
      id: TEST_IDS.ESCOLA,
      nome: 'Escola E2E Teste',
      tipo: 'pre_escola',
      ativo: true,
    }, { onConflict: 'id' })
    .select()
    .single()

  if (error) {
    console.error('Failed to seed test escola:', error)
    throw error
  }

  return data
}

/**
 * Seed a test turma
 */
export async function seedTestTurma() {
  const { data, error } = await supabaseAdmin
    .from('turmas')
    .upsert({
      id: TEST_IDS.TURMA,
      nome: 'Turma E2E Teste',
      serie: 'Pre I',
      ano_letivo: new Date().getFullYear(),
      escola_id: TEST_IDS.ESCOLA,
      turno: 'matutino',
      capacidade: 25,
      ativo: true,
    }, { onConflict: 'id' })
    .select()
    .single()

  if (error) {
    console.error('Failed to seed test turma:', error)
    throw error
  }

  return data
}

/**
 * Cleanup all test data
 */
export async function cleanupTestData() {
  // Delete in reverse order of dependencies
  const tables = [
    { table: 'turmas', id: TEST_IDS.TURMA },
    { table: 'escolas', id: TEST_IDS.ESCOLA },
  ]

  for (const { table, id } of tables) {
    const { error } = await supabaseAdmin
      .from(table)
      .delete()
      .eq('id', id)

    if (error) {
      console.warn(`Failed to cleanup ${table}:`, error.message)
    }
  }
}

/**
 * Playwright Authentication Setup
 *
 * Authenticates as different user roles and caches the auth state.
 * Each role's auth state is saved to a separate JSON file.
 *
 * Per RESEARCH.md: "Cache auth state per role (admin.json, professor.json)"
 */

import { test as setup, expect } from '@playwright/test'
import path from 'path'
import { getTestCredentials } from './helpers/seed'

const AUTH_DIR = path.join(__dirname, '../../playwright/.auth')

// Helper function to perform login
async function performLogin(page: any, email: string, password: string) {
  await page.goto('/login')

  // Wait for login form to be visible
  await page.waitForSelector('input[type="email"], input[id="email"], input[name="email"]', { timeout: 10000 })

  // Fill in credentials
  await page.fill('input[type="email"], input[id="email"], input[name="email"]', email)
  await page.fill('input[type="password"], input[id="password"], input[name="password"]', password)

  // Submit form
  await page.click('button[type="submit"]')

  // Wait for successful navigation to dashboard
  await expect(page).toHaveURL(/\/dashboard/, { timeout: 15000 })
}

// Setup: Authenticate as admin
setup('authenticate as admin', async ({ page }) => {
  const credentials = getTestCredentials('admin')

  try {
    await performLogin(page, credentials.email, credentials.password)

    // Save storage state for reuse
    await page.context().storageState({ path: path.join(AUTH_DIR, 'admin.json') })
  } catch (error) {
    console.error('Admin auth setup failed:', error)
    throw error
  }
})

// Setup: Authenticate as professor
setup('authenticate as professor', async ({ page }) => {

/**
 * Login Flow E2E Tests
 *
 * Tests the authentication flow for the EDUCA system.
 * Covers: valid login, invalid credentials, form validation.
 *
 * Requirement: MWE-06 - E2E tests cover login and authentication flow
 */

import { test, expect } from '@playwright/test'

test.describe('Login Flow', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/login')
  })

  test('shows login form with required fields', async ({ page }) => {
    // Verify login form elements are present
    await expect(page.locator('input[type="email"], input[id="email"]')).toBeVisible()
    await expect(page.locator('input[type="password"], input[id="password"]')).toBeVisible()
    await expect(page.locator('button[type="submit"]')).toBeVisible()
  })

  test('shows validation error for empty email', async ({ page }) => {
    // Try to submit with empty email
    await page.fill('input[type="password"], input[id="password"]', 'somepassword')
    await page.click('button[type="submit"]')

    // Should show validation message or stay on login page
    await expect(page).toHaveURL(/\/login/)
  })

  test('shows validation error for empty password', async ({ page }) => {
    // Try to submit with empty password
    await page.fill('input[type="email"], input[id="email"]', 'test@example.com')
    await page.click('button[type="submit"]')

    // Should show validation message or stay on login page
    await expect(page).toHaveURL(/\/login/)
  })

  test('shows error message for invalid credentials', async ({ page }) => {
    // Submit with invalid credentials
    await page.fill('input[type="email"], input[id="email"]', 'invalid@test.com')
    await page.fill('input[type="password"], input[id="password"]', 'wrongpassword')
    await page.click('button[type="submit"]')

    // Should stay on login page and show error
    await expect(page).toHaveURL(/\/login/)

    // Look for error message (toast or inline)
    // The exact selector depends on how errors are shown
    const errorVisible = await page.locator('[role="alert"], .error-message, [data-sonner-toast]').isVisible().catch(() => false)

    // If no visible error element, at least verify we're still on login
    if (!errorVisible) {
      await expect(page).toHaveURL(/\/login/)
    }
  })

  test('redirects to dashboard on successful login', async ({ page }) => {
    // This test requires valid test credentials
    // Skip if credentials are not available
    const email = process.env.E2E_ADMIN_EMAIL
    const password = process.env.E2E_ADMIN_PASSWORD

    if (!email || !password) {
      test.skip(true, 'E2E credentials not configured')
    }

    await page.fill('input[type="email"], input[id="email"]', email!)
    await page.fill('input[type="password"], input[id="password"]', password!)
    await page.click('button[type="submit"]')

    // Should redirect to dashboard
    await expect(page).toHaveURL(/\/dashboard/, { timeout: 15000 })
  })

  test('preserves return URL after login', async ({ page }) => {
    // Try to access protected page directly
    await page.goto('/dashboard/turmas')

    // Should redirect to login (if not authenticated)
    // After login, should return to original URL

    // This test verifies the redirect behavior
    // The exact behavior depends on auth middleware implementation
    const currentUrl = page.url()
    expect(currentUrl).toMatch(/\/(login|dashboard)/)
  })

  test('logout returns to login page', async ({ page, context }) => {
    // This test requires being logged in first
    const email = process.env.E2E_ADMIN_EMAIL
    const password = process.env.E2E_ADMIN_PASSWORD

    if (!email || !password) {
      test.skip(true, 'E2E credentials not configured')
    }

    // Login first
    await page.fill('input[type="email"], input[id="email"]', email!)
    await page.fill('input[type="password"], input[id="password"]', password!)
    await page.click('button[type="submit"]')
    await expect(page).toHaveURL(/\/dashboard/, { timeout: 15000 })

    // Find and click logout button
    // This may be in a dropdown menu
    const profileButton = page.locator('button:has-text("U"), [aria-label*="profile"], [aria-label*="user"]').first()
    if (await profileButton.isVisible()) {
      await profileButton.click()
    }

    const logoutButton = page.locator('button:has-text("Sair"), [role="menuitem"]:has-text("Sair")')
    if (await logoutButton.isVisible()) {
      await logoutButton.click()
    }

    // Should redirect to login
    await expect(page).toHaveURL(/\/login/, { timeout: 10000 })
  })
})

test.describe('Protected Routes', () => {
  test('redirects unauthenticated users from dashboard', async ({ page }) => {
    // Clear any existing auth state
    await page.context().clearCookies()

    // Try to access dashboard directly
    await page.goto('/dashboard')

    // Should redirect to login
    await expect(page).toHaveURL(/\/login/)
  })

  test('redirects unauthenticated users from turmas page', async ({ page }) => {
    await page.context().clearCookies()
    await page.goto('/dashboard/turmas')
    await expect(page).toHaveURL(/\/login/)
  })

  test('redirects unauthenticated users from chamada page', async ({ page }) => {
    await page.context().clearCookies()
    await page.goto('/dashboard/chamada')
    await expect(page).toHaveURL(/\/login/)
  })
})
