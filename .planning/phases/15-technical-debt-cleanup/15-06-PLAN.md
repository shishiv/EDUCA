   ```typescript
   async calculateFrequency(matriculaId: string, dateRange?: DateRange): Promise<number> {
     // Fetch all attendance records for the matricula
     const { data: records } = await supabase
       .from('frequencia')
       .select('presente')
       .eq('matricula_id', matriculaId)
       .gte('data', dateRange?.start || firstDayOfMonth)
       .lte('data', dateRange?.end || today)

     if (!records || records.length === 0) return 0

     const present = records.filter(r => r.presente).length
     const total = records.length
     return Math.round((present / total) * 100)
   }
   ```
